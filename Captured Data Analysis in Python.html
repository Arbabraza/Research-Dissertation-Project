records = []

for pkt in packets:
    if IP in pkt:
        proto = "TCP" if TCP in pkt else "UDP" if UDP in pkt else "OTHER"

        records.append({
            "src_ip": pkt[IP].src,
            "dst_ip": pkt[IP].dst,
            "length": len(pkt),
            "protocol": proto,
            "ttl": pkt[IP].ttl
        })

df = pd.DataFrame(records)
